void automation.AutoAssignAgentUpdateInEvery30Minutes(Int leadVerificationId)
{
LeadVerificationData = zoho.crm.getRecordById("Lead_Verifications",leadVerificationId);
is_accepted = LeadVerificationData.get("Accepted");
status = LeadVerificationData.get("Status");
if(is_accepted == true || status == null || status == "" || status == "Skipped (Moved to Next Agent)")
{
	LeadVerificationData = zoho.crm.getRecordById("Lead_Verifications",leadVerificationId);
	lead_id = LeadVerificationData.get("Lead_Name").get("id");
	campaign_id = LeadVerificationData.get("Campaign_Name").get("id");
	team_name = LeadVerificationData.get("Team");
	dev_id = LeadVerificationData.get("Dev_Id") + 1;
	info dev_id;
	email = LeadVerificationData.get("Email");
	searchDataAgents = zoho.crm.searchRecords("Team_Member","(Teams:equals:" + team_name + ")");
	info searchDataAgents;
	new_agent_id = searchDataAgents.get(dev_id).get("Agent_CRM_User_ID");
	new_agent_name = searchDataAgents.get(dev_id).get("Name");
	valid_time = zoho.currenttime.addMinutes(1).toString("yyyy-MM-dd'T'HH:mm:ssXXX");
	putMap = Map();
	putMap.put("Valid_Till",valid_time);
	putMap.put("Dev_Id",dev_id);
	putMap.put("Lead_Name",lead_id);
	putMap.put("Campaign_Name",campaign_id);
	putMap.put("Owner",new_agent_id);
	putMap.put("Name","Lead Verification for: " + new_agent_name);
	putMap.put("Team",team_name);
	putMap.put("Email",email);
	createRecord = zoho.crm.createRecord("Lead_Verifications",putMap);
	leadData = Map();
	info createRecord;
}
else
{
	leadData.put("Owner",new_agent_id);
	updateLead = zoho.crm.updateRecord("Leads",lead_id,leadData);
	info updateLead;
}
}
